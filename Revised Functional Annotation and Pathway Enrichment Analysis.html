Phase 1: Download & File Preparation

Prepare your working directory and ensure all mandatory source files are organized before beginning the pipeline.

Required Reference Files and Directories:

  • trinity_out.Trinity.fasta: The de novo assembled transcripts.
  • trinity_out.Trinity.fasta.gene_trans_map: Links transcript IDs to corresponding gene IDs.
  • Trinity.transdecoder.pep: Predicted protein sequences from TransDecoder.
  • conditionA1 to B3: Folders containing raw data (FASTQ) or aligned reads.
  • deseq_inputs: Directory housing formatted datasets for downstream differential analysis.
  • databases: Directory containing reference databases (e.g., UniProt, Pfam, Trinotate.sqlite).
  • transdecoder: Directory with intermediate files from translation steps.

Note: Remember to make a directory for your databases (Pfam, Trinotate.sqlite, etc.) before running annotations.

Phase 3: Sequence Alignment and Search (DIAMOND & HMMER)

Run sequence-level similarity searches to uncover alignments in protein databases. 

#!/bin/bash
# Save as diamond.sh and execute
# Make sure to adjust paths inside your 'databases/' folder

PEP_FILE="Trinity.transdecoder.pep"
UNIPROT_DB="databases/uniprot.dmnd" 
PFAM_DB="databases/Pfam-A.hmm"
THREADS=4 

# 1. Run DIAMOND blastp against UniProt/SwissProt
echo "Running DIAMOND blastp against UniProt..."
diamond blastp \
    --query "$PEP_FILE" \
    --db "$UNIPROT_DB" \
    --out blastp_uniprot.outfmt6 \
    --outfmt 6 \
    --evalue 1e-5 \
    --max-target-seqs 1 \
    --threads "$THREADS"

# 2. Run hmmscan against Pfam
echo "Running hmmscan against Pfam..."
hmmscan \
    --cpu "$THREADS" \
    --domtblout pfam.domtblout \
    "$PFAM_DB" \
    "$PEP_FILE" > pfam.log

echo "Searches complete!"

Checklist for aligner outputs:

  • pfam.domtblout
  • pfam.log
  • blastp_uniprot.outfmt6

Step 4.3: Cleaning and Mapping IDs (R Script)

Resolve mapping conflicts: eggNOG-mapper uses protein IDs, while Trinotate works with Transcript/Gene IDs. 

library(dplyr)

# 1. Load the gene-to-transcript map
gene_map <- read.delim("trinity_out.Trinity.fasta.gene_trans_map", 
                       header=FALSE, 
                       col_names=c("transcript_id", "gene_id"))

# 2. Load eggNOG with manual headers
col_names <- c("query", "seed_ortholog", "evalue", "score", "eggNOG_OGs", 
               "max_annot_lvl", "COG_category", "Description", "Preferred_name", 
               "GOs", "EC", "KEGG_ko", "KEGG_Pathway", "KEGG_Module", 
               "KEGG_Reaction", "KEGG_rclass", "BRITE", "KEGG_TC", 
               "CAZy", "BiGG_Reaction", "PFAMs")

eggnog <- read.table("annotation_results.emapper.annotations", 
                     sep = "\t", 
                     comment.char = "#", 
                     header = FALSE, 
                     col.names = col_names,
                     check.names = FALSE)

# Clean eggNOG structure
eggnog_clean <- eggnog %>% 
  select(gene_id = query, KO_ID = KEGG_ko)

# Clean protein isoforms (strip trailing '.p1', etc.) to map to standard transcript IDs
eggnog_clean$transcript_id <- gsub("\\.p[0-9]+$", "", eggnog_clean$gene_id)

# Merge eggNOG with gene_map
eggnog_mapped <- eggnog_clean %>%
  left_join(gene_map, by = "transcript_id")

# Load Trinotate
trinotate <- read.delim("final_trinotate_report.xls", sep="\t")

# Final master annotation build
master_annotation <- trinotate %>%
  left_join(eggnog_mapped, by = c("X.gene_id" = "transcript_id"))

# Export mappings
write.csv(master_annotation, "gene_to_KO_mapping.csv", row.names = FALSE)

Step 4.4: Integrated Master Annotation

This is where we will integrate everything: DGE statistics (DESeq2), KO/GO mappings, and the Trinotate report to create the complete file. 

library(dplyr)
library(stringr)

# 1. Load DESeq2 Results
deseq_res <- read.csv("DESeq2_full_results.csv")

# 2. Function to safely load mapping files
load_map <- function(filename) {
  raw_lines <- readLines(filename)
  split_data <- str_split_fixed(raw_lines, "\t", 2)
  df <- as.data.frame(split_data, stringsAsFactors = FALSE)
  colnames(df) <- c("feature_id", "ID")
  df <- df %>% filter(feature_id != "") %>% mutate(across(everything(), str_trim))
  return(df)
}

# 3. Load KO and GO files
ko_map <- load_map("user_ko.txt")
colnames(ko_map)[2] <- "KO_id"
go_map <- load_map("gene_ontology_annotations.txt")
colnames(go_map)[2] <- "GO_id"

ko_map$clean_id <- gsub("\\.p[0-9]+$", "", ko_map$feature_id)
go_map$clean_id <- gsub("\\.p[0-9]+$", "", go_map$feature_id)

# 4. Load Trinotate Report
trinotate <- read.delim("final_trinotate_report.xls", sep="\t", header=TRUE, fill=TRUE, quote="", stringsAsFactors=FALSE)
trinotate$clean_id <- gsub("\\.p[0-9]+$", "", trinotate$X.gene_id)

# 5. Integrate everything into one Master Table
master_df <- deseq_res %>%
  left_join(ko_map %>% select(clean_id, KO_id), by = c("feature_id" = "clean_id")) %>%
  left_join(go_map %>% select(clean_id, GO_id), by = c("feature_id" = "clean_id")) %>%
  left_join(trinotate %>% select(clean_id, sprot_Top_BLASTX_hit, Pfam, eggnog, Kegg), 
            by = c("feature_id" = "clean_id"))

# 6. Save Files
write.csv(master_df, "Master_Annotation_Complete.csv", row.names = FALSE)

annotated_only <- master_df %>% filter(!is.na(KO_id) | !is.na(GO_id))
write.csv(annotated_only, "Annotated_Functional_DEGs.csv", row.names = FALSE)

message("Master Annotation and Functional DEG table created successfully.")

Phase 5: Differential Expression and Pathway Analysis (DESeq2)

Perform the downstream calculations in RStudio using the reproducible script. 

# --- Internship Activity - Transcriptome Analysis Workflow (FINAL & UNIFIED) ---

# ==========================================
# 0. LOAD REQUIRED LIBRARIES
# ==========================================
library(DESeq2); library(clusterProfiler); library(ggplot2); library(readr); library(tibble)
library(pheatmap); library(tidyr); library(enrichplot); library(pathview); library(AnnotationDbi)
library(dplyr); library(stringr); library(KEGGREST); library(GO.db)

# Configuration
output_dir <- "D:/deseqinputs/analysis_results"
dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)

# ==========================================
# Step 1 & 2: DESeq2 Analysis
# ==========================================
message("Running DESeq2...")
count_matrix <- read.csv("count_matrix.csv", row.names = 1)
metadata <- read.csv("metadata.csv", row.names = 1)
metadata$group1 <- relevel(as.factor(metadata$Treatment), ref = "conditionA")

dds <- DESeqDataSetFromMatrix(countData = count_matrix, colData = metadata, design = ~ group1)
dds <- DESeq(dds, test = "Wald")
res <- results(dds, contrast = c("group1", "conditionB", "conditionA"))
res_df <- as.data.frame(res) %>% rownames_to_column("feature_id") %>%
  mutate(direction = case_when(pvalue < 0.05 & log2FoldChange > 0.5 ~ "Up",
                               pvalue < 0.05 & log2FoldChange < -0.5 ~ "Down",
                               TRUE ~ "Not_DE"))
res_df$feature_id <- gsub("\\.p[0-9]+$", "", res_df$feature_id)

# ==========================================
# Step 3: Master Annotation & Mapping Prep
# ==========================================
message("Preparing Annotations and Master Table...")

# Safe file loading function
load_map <- function(filename) {
  raw_lines <- readLines(filename)
  split_data <- str_split_fixed(raw_lines, "\t", 2)
  df <- as.data.frame(split_data, stringsAsFactors = FALSE)
  colnames(df) <- c("feature_id", "ID")
  df <- df %>% filter(feature_id != "") %>% mutate(across(everything(), str_trim))
  return(df)
}

# Load mappings
ko_map <- load_map("user_ko.txt")
colnames(ko_map) <- c("feature_id", "KO_id")
go_map <- load_map("gene_ontology_annotations.txt")
colnames(go_map) <- c("feature_id", "GO_id")

ko_map$clean_id <- gsub("\\.p[0-9]+$", "", ko_map$feature_id)
go_map$clean_id <- gsub("\\.p[0-9]+$", "", go_map$feature_id)

# Load Trinotate
trinotate <- read.delim("final_trinotate_report.xls", sep="\t", header=TRUE, fill=TRUE, quote="", stringsAsFactors=FALSE)
trinotate$clean_id <- gsub("\\.p[0-9]+$", "", trinotate$X.gene_id)

# Create Master Annotation Table
master_df <- res_df %>%
  left_join(ko_map %>% select(clean_id, KO_id), by = c("feature_id" = "clean_id")) %>%
  left_join(go_map %>% select(clean_id, GO_id), by = c("feature_id" = "clean_id")) %>%
  left_join(trinotate %>% select(clean_id, sprot_Top_BLASTX_hit, Pfam, eggnog, Kegg), 
            by = c("feature_id" = "clean_id"))

write.csv(master_df, file.path(output_dir, "Master_Annotation_Complete.csv"), row.names = FALSE)

# Prepare mapping for enricher
ko_to_path <- bitr_kegg(unique(ko_map$KO_id), fromType="kegg", toType="Path", organism="ko")
path_to_gene <- merge(ko_to_path, ko_map, by.x="kegg", by.y="KO_id")
kegg_term2gene <- path_to_gene[, c("Path", "feature_id")]
go_term2gene <- go_map[, c("GO_id", "clean_id")]

# Prepare Gene Lists
genes_up <- res_df$feature_id[res_df$direction == "Up"]
genes_down <- res_df$feature_id[res_df$direction == "Down"]
genes_all <- res_df$feature_id[res_df$direction %in% c("Up", "Down")]

# ==========================================
# Step 4: Individual GO and KEGG Plots (WITH PROPER NAMES)
# ==========================================
message("Generating Enrichment Plots...")

# --- GO ALL PLOT ---
go_res <- enricher(gene = genes_all, TERM2GENE = go_term2gene, pvalueCutoff = 1, qvalueCutoff = 1)
if (!is.null(go_res) && nrow(go_res@result) > 0) {
  raw_go_ids <- as.character(go_res@result$ID)
  clean_go_ids <- str_extract(raw_go_ids, "GO:[0-9]{7}")
  go_names <- mapIds(GO.db, keys=clean_go_ids, column="TERM", keytype="GOID", multiVals="first")
  go_names[is.na(go_names)] <- raw_go_ids[is.na(go_names)]
  go_res@result$Description <- unname(go_names)
  
  p_go <- dotplot(go_res, showCategory=15) + ggtitle("Dot Plot: GO_All") +
    scale_y_discrete(labels = function(x) str_wrap(x, width = 45)) + theme(axis.text.y = element_text(size = 9))
  ggsave(file.path(output_dir, "GO_All_DotPlot_Final.png"), p_go, width=8, height=8)
}

# --- FETCH KEGG NAMES ---
kegg_info <- keggList("pathway", "ko")
names(kegg_info) <- str_replace(names(kegg_info), "path:", "")

# --- KEGG UP PLOT ---
kegg_up <- enricher(gene = genes_up, TERM2GENE = kegg_term2gene, pvalueCutoff = 1, qvalueCutoff = 1)
if (!is.null(kegg_up) && nrow(kegg_up@result) > 0) {
  up_ids <- as.character(kegg_up@result$ID)
  up_names <- unname(kegg_info[up_ids])
  up_names[is.na(up_names)] <- up_ids[is.na(up_names)]
  kegg_up@result$Description <- up_names
  
  p_up <- dotplot(kegg_up, showCategory=15) + ggtitle("Dot Plot: KEGG_Up") +
    scale_y_discrete(labels = function(x) str_wrap(x, width = 45)) + theme(axis.text.y = element_text(size = 9))
  ggsave(file.path(output_dir, "KEGG_Up_DotPlot_Final.png"), p_up, width=8, height=8)
}

# --- KEGG DOWN PLOT ---
kegg_down <- enricher(gene = genes_down, TERM2GENE = kegg_term2gene, pvalueCutoff = 1, qvalueCutoff = 1)
if (!is.null(kegg_down) && nrow(kegg_down@result) > 0) {
  down_ids <- as.character(kegg_down@result$ID)
  down_names <- unname(kegg_info[down_ids])
  down_names[is.na(down_names)] <- down_ids[is.na(down_names)]
  kegg_down@result$Description <- down_names
  
  p_down <- dotplot(kegg_down, showCategory=15) + ggtitle("Dot Plot: KEGG_Down") +
    scale_y_discrete(labels = function(x) str_wrap(x, width = 45)) + theme(axis.text.y = element_text(size = 9))
  ggsave(file.path(output_dir, "KEGG_Down_DotPlot_Final.png"), p_down, width=8, height=8)
}

# --- PAIRWISE KEGG PLOT ---
gene_list_split <- list(Up = genes_up, Down = genes_down)
kegg_term2name <- ko2name(unique(kegg_term2gene$Path))
colnames(kegg_term2name) <- c("Path", "Name")
kegg_term2name$Name <- paste0(kegg_term2name$Path, " (", kegg_term2name$Name, ")")

ck <- compareCluster(geneCluster = gene_list_split, fun = "enricher", TERM2GENE = kegg_term2gene,
                     TERM2NAME = kegg_term2name, pvalueCutoff = 1, qvalueCutoff = 1)
ggsave(file.path(output_dir, "Pairwise_KEGG_Dotplot.png"), dotplot(ck, showCategory=20) + ggtitle("Pairwise KEGG: Pathway & ID Integration"))

# ==========================================
# Step 5: Final Pathview Mapping
# ==========================================
message("Generating Pathview maps...")
lfc_ko <- setNames(res_df$log2FoldChange[match(ko_map$feature_id, res_df$feature_id)], ko_map$KO_id)
lfc_ko <- lfc_ko[!is.na(names(lfc_ko))]

sig_df <- as.data.frame(ck)
valid_paths <- unique(str_extract(sig_df$ID, "[0-9]{5}")[!is.na(str_extract(sig_df$ID, "[0-9]{5}"))])

if(length(valid_paths) > 0) {
  for (pid in head(valid_paths, 3)) { 
    tryCatch({
      pathview(gene.data = lfc_ko, pathway.id = pid, species = "ko", gene.idtype = "KEGG", dest.dir = output_dir, kegg.native = TRUE)
    }, error = function(e) message("Pathview failed for ", pid, ": ", e$message))
  }
}

# ==========================================
# Step 6: Volcano Plot
# ==========================================
res_df$sig <- ifelse(res_df$pvalue < 0.05 & abs(res_df$log2FoldChange) > 0.5, "Significant", "Not Significant")
p_volcano <- ggplot(res_df, aes(x=log2FoldChange, y=-log10(pvalue), color=sig)) + 
  geom_point(alpha=0.6) + theme_minimal() + ggtitle("Volcano Plot")
ggsave(file.path(output_dir, "Volcano_Plot.png"), p_volcano)

message("Workflow Complete! All fixed plots and master tables are saved in: ", output_dir)
